1:     create an adjacency matrix m with order n;
     2:     let nRow = the smallest number in each row;
     3:     subtract nRow from all elements in the row;
     4:     let nCol = the smallest number in each column;
     5:     subtract nCol from all elements in the column;
     6:     let l = the minimal number of vertical and/or horizontal lines which can cover all zeroes;
     7:           while l is smaller than n
     8:                 let s = the smallest uncovered number;
     9:                 subtract s from all uncovered numbers and it to all cross-covered numbers;
    10:                 l = new vertical and/or horizontal lines which can cover all zeroes;
    11:     the optimal assignment is found
